57 Matrix Matrix::Inverse() const {
58 
59  float d = 0;
60 
61  // Left half of the augmented matrix in gaussian elimination
62  Matrix L = Matrix(values);
63 
64  // Right half of the augmented matrix (identity matrix) in gaussian elimination
65  Matrix R = Matrix(dim.lines, dim.cols);
66  for (size_t i = 0; i < dim.lines; i++)
67  R[i][i] = 1;
68 
69  // Partial pivoting
70  for (size_t i = dim.lines - 1; i > 0; i--) {
71  if (L[i - 1][0] < L[i][0])
72  for (size_t j = 0; j < dim.lines; j++) {
73  d = L[i][j];
74  L[i][j] = L[i - 1][j];
75  L[i - 1][j] = d;
76 
77  d = R[i][j];
78  R[i][j] = R[i - 1][j];
79  R[i - 1][j] = d;
80  }
81  }
82 
83  //cout << "pivoted output: " << endl;
84  //cout << L << endl;
85  //cout << R << endl;
86 
87  // reduce to diagonal matrix
88 
89  for (size_t i = 0; i < dim.lines; i++) {
90  for (size_t j = 0; j < dim.lines; j++)
91  if (j != i) {
92  d = L[j][i] / L[i][i];
93 
94  if(isnan(d)) Exceptions::InvertibleException();
95 
96  for (size_t k = 0; k < dim.lines; k++) {
97  R[j][k] -= R[i][k] * d;
98  L[j][k] -= L[i][k] * d;
99  }
100  }
101  }
102 
103  //cout << "diag " << endl;
104  //cout << L << endl;
105  //cout << R << endl;
106 
107  // reduce to identity matrix
108  for (size_t i = 0; i < dim.lines; i++) {
109  d = L[i][i];
110  for (size_t j = 0; j < dim.lines; j++)
111  R[i][j] = R[i][j] / d;
112  }
113 
114  return R;
115 }
